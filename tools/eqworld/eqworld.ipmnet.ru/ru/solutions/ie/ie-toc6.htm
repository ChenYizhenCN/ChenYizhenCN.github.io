6. Нелинейные интегральные уравнения с постоянными пределами интегрирования

  1. g(t)y(x)y(t) dt = f(x).
  2. f(t)y(t)y(xt) dt = A.
  3. f(t)y(t)y(x/t) dt = Axλ.
  4. y(x) + g(t)y(x)y(t) dt = f(x).
  5. y(x) + g(x)y(x)y(t) dt = f(x).
  6. y(x) + f(t)y(t)y(x/t) dt = 0.
  7. y(x) + f(t)y(x/t)y(t) dt = Axb.
  8. y(x) + f(t, y(t)) dt = g(x).
  9. y(x) + eλ(x − t)f(t, y(t)) dt = g(x).
  10. y(x) + g(x)f(t, y(t)) dt = h(x).
  11. y(x) + |x − t|f(t, y(t)) dt = g(x).
  12. y(x) + eλ|x − t|f(t, y(t)) dt = g(x).
